package org.marketplace_lea.order.domain.order.services.impl;

import com.conso_epargne.common_service.common.exceptions.ConsoEpargneException;
import com.conso_epargne.common_service.entities.PaymentDetail;
import com.conso_epargne.common_service.entities.PaymentMethod;
import com.conso_epargne.common_service.entities.Voucher;
import com.conso_epargne.common_service.entities.v2.order.OrderV2Entity;
import com.conso_epargne.common_service.repositories.PaymentDetailRepository;
import com.conso_epargne.common_service.repositories.VoucherRepository;
import org.marketplace_lea.order.domain.order.dto.CreateOrderV2Form;
import org.marketplace_lea.order.domain.order.services.PaymentDetailService;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.http.HttpStatus;
import org.springframework.stereotype.Service;

import java.util.Date;
import java.util.Optional;

@Slf4j
@Service
@RequiredArgsConstructor
public class DefaultPaymentDetailService implements PaymentDetailService {

    private final PaymentDetailRepository paymentDetailRepository;
    private final VoucherRepository voucherRepository;

    @Override
    public PaymentDetail createPaymentDetailForOrder(PaymentMethod paymentMethod, OrderV2Entity order, CreateOrderV2Form createDTO) {
        log.info("[DefaultPaymentDetailService.createPaymentDetailForOrder] Creating payment details for order: {}", order.getId());

        // Calculate delivery fees from order
        float deliveryFees = order.getDeliveryFees();

        // Build payment details
        var paymentDetails = PaymentDetail.builder()
                .paymentMethod(paymentMethod)
                .deliveryFees(deliveryFees)
                .amount(deliveryFees + order.getTotalAmount())
                .build();

        // Calculate payment fees
        float paymentFees = paymentDetails.calculateAmountWithFees(paymentMethod.getFees());
        paymentDetails.setAmount(paymentFees + paymentDetails.getAmount());

        // Handle voucher if provided
        var voucher = getCorrectVoucher(paymentDetails, createDTO);
        paymentDetails.setVoucher(voucher);

        // Set online fees for mobile money payments
        if (paymentMethod.isMobileMoneyPayment()) {
            paymentDetails.setOnlineFees(paymentFees);
        }

        var paymentDetailsSaved = paymentDetailRepository.save(paymentDetails);
        log.info("[DefaultPaymentDetailService.createPaymentDetailForOrder] Payment details saved: {}", paymentDetailsSaved.getId());
        return paymentDetailsSaved;
    }

    private Voucher getCorrectVoucher(PaymentDetail paymentDetail, CreateOrderV2Form createDTO) {
        Optional<Voucher> voucher;
        if (createDTO.voucherId() != null && !createDTO.voucherId().isBlank()) {
            voucher = voucherRepository.findById(createDTO.voucherId());

            if (voucher.isEmpty()) {
                throw new ConsoEpargneException("Le bon d'achat est introuvable !", HttpStatus.BAD_REQUEST);
            }

            var voucherFound = voucher.get();
            float amount = paymentDetail.getAmount() - voucherFound.getAmount();
            amount = amount < 0 ? amount : 0;
            paymentDetail.setAmount(amount);
            voucherFound.setValid(false);
            voucherFound.setExpiredAt(new Date());
            return voucherRepository.save(voucherFound);
        }

        return null;
    }
}
