package org.marketplace_lea.order.domain.order.services;

import com.conso_epargne.common_service.entities.PaymentMethod;
import com.conso_epargne.common_service.entities.v2.order.OrderV2Entity;
import org.marketplace_lea.order.domain.order.dto.CreateOrderV2Form;

/**
 * Service for handling payment details for orders.
 * Based on PaymentDetailsService from common-service, adapted for v2 entities.
 */
public interface PaymentDetailService {
    
    /**
     * Creates payment details for an order based on payment method and order form.
     * 
     * @param paymentMethod The payment method to use
     * @param order The order entity
     * @param createDTO The order creation form
     * @return The created payment detail
     */
    com.conso_epargne.common_service.entities.PaymentDetail createPaymentDetailForOrder(
            PaymentMethod paymentMethod, 
            OrderV2Entity order, 
            CreateOrderV2Form createDTO);
}
